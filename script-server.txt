INSTALAR NGINX

sudo nano /etc/nginx/sites-available/jselectronics.live

server {
    listen 80;
    server_name jselectronics.live www.jselectronics.live;

    location / {
        proxy_pass http://localhost:3000;  # Proxy traffic to Node.js on port 3000
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

sudo ln -s /etc/nginx/sites-available/jselectronics.live /etc/nginx/sites-enabled/

sudo nginx -t
sudo systemctl reload nginx
